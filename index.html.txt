    const adModal = document.getElementById("adModal");
    const adNumberDisplay = document.getElementById("adNumber");
    const adTimerDisplay = document.getElementById("adTimer");
    const adStatusDisplay = document.getElementById("adStatus");
    const adViewingMessage = document.getElementById("adViewingMessage");
    const mainContent = document.getElementById("mainContent");
    const startContainer = document.getElementById("start-container");
    const adStartSound = document.getElementById("adStartSound");
    const adSuccessSound = document.getElementById("adSuccessSound");
    const hoverSound = document.getElementById("hoverSound");
    const startButton = document.getElementById("start-ads");
    const downloadLinks = document.querySelectorAll(".download-box a");
    const socialButtons = document.querySelectorAll(".social-button");

    const adLinks = [
      "https://otieu.com/4/10024540",
      "https://otieu.com/4/10024543",
      "https://otieu.com/4/10024550",
      "https://otieu.com/4/10024551"
    ];
    let currentAdIndex = 0;
    let adViewCount = 0;
    const adViewTime = 10;

    // GSAP Animations for disabled buttons
    const disabledTween = gsap.to(".download-box a:not(.enabled)", {
      scale: 1.03,
      opacity: 0.7,
      repeat: -1,
      yoyo: true,
      duration: 1,
      ease: "power1.inOut"
    });

    gsap.fromTo("#start-ads", 
      { scale: 1, boxShadow: "0 4px 15px rgba(244, 67, 54, 0.5)" },
      { scale: 1.1, boxShadow: "0 8px 20px rgba(244, 67, 54, 0.7)", repeat: -1, yoyo: true, duration: 1.5, ease: "power1.inOut" }
    );

    function animateEnabledButtons() {
      gsap.to(".download-box a.enabled", {
        scale: 1.05,
        rotate: 2,
        duration: 0.5,
        ease: "elastic.out(1, 0.3)",
        stagger: 0.2,
        onComplete: () => {
          gsap.to(".download-box a.enabled", {
            backgroundPosition: "200% 0",
            duration: 3,
            repeat: -1,
            ease: "none"
          });
        }
      });

      downloadLinks.forEach(link => {
        link.addEventListener("mouseenter", () => {
          if (link.classList.contains("enabled")) {
            gsap.to(link, {
              scale: 1.15,
              y: -5,
              boxShadow: "0 10px 25px rgba(76, 175, 80, 0.8)",
              borderColor: "#81c784",
              duration: 0.3,
              ease: "power2.out"
            });
            gsap.to(link.querySelector("span"), {
              y: -3,
              scale: 1.1,
              duration: 0.3,
              ease: "bounce.out"
            });
            hoverSound.currentTime = 0;
            hoverSound.play().catch(error => console.log("Error playing hover sound:", error));
          }
        });
        link.addEventListener("mouseleave", () => {
          if (link.classList.contains("enabled")) {
            gsap.to(link, {
              scale: 1.05,
              y: 0,
              boxShadow: "0 4px 15px rgba(76, 175, 80, 0.5)",
              borderColor: "#4CAF50",
              duration: 0.3,
              ease: "power2.out"
            });
            gsap.to(link.querySelector("span"), {
              y: 0,
              scale: 1,
              duration: 0.3,
              ease: "power2.out"
            });
          }
        });
      });
    }

    // Add hover animations to social buttons
    socialButtons.forEach(button => {
      button.addEventListener("mouseenter", () => {
        gsap.to(button, {
          scale: 1.1,
          boxShadow: "0 6px 12px rgba(0, 0, 0, 0.3)",
          duration: 0.3,
          ease: "power2.out"
        });
        hoverSound.currentTime = 0;
        hoverSound.play().catch(error => console.log("Error playing hover sound:", error));
      });
      button.addEventListener("mouseleave", () => {
        gsap.to(button, {
          scale: 1,
          boxShadow: "none",
          duration: 0.3,
          ease: "power2.out"
        });
      });
    });

    downloadLinks.forEach(link => {
      link.classList.remove("enabled");
      link.style.background = "linear-gradient(45deg, #2196f3, #64b5f6)";
    });

    function showAdModal() {
      gsap.to(adModal, { opacity: 1, duration: 0.5, onStart: () => adModal.classList.remove("hidden") });
      document.body.style.overflow = "hidden";
      adNumberDisplay.textContent = currentAdIndex + 1;
      adViewingMessage.textContent = `You are viewing Ad ${currentAdIndex + 1} of 4`;
      adStatusDisplay.textContent = "";
      startAdTimer();
    }

    function startAdTimer() {
      let timeLeft = adViewTime;
      adTimerDisplay.textContent = `Time remaining: ${timeLeft} seconds`;

      adStartSound.currentTime = 0;
      adStartSound.play().catch(error => console.log("Error playing ad start sound:", error));

      const adWindow = window.open(adLinks[currentAdIndex], "_blank");

      const timer = setInterval(() => {
        timeLeft--;
        adTimerDisplay.textContent = `Time remaining: ${timeLeft} seconds`;

        if (timeLeft <= 0) {
          clearInterval(timer);
          adViewingMessage.textContent = `Ad ${currentAdIndex + 1} viewed! You can proceed.`;
          adStatusDisplay.textContent = `Ad ${currentAdIndex + 1} completed.`;

          adSuccessSound.currentTime = 0;
          adSuccessSound.play().catch(error => console.log("Error playing ad success sound:", error));

          adViewCount++;
          currentAdIndex++;

          if (adWindow && adWindow.closed) {
            adViewingMessage.textContent = `Ad ${currentAdIndex} was closed early. Please view again.`;
            adStatusDisplay.textContent = "";
            currentAdIndex--;
            setTimeout(showAdModal, 1000);
            return;
          }

          if (adWindow) adWindow.close();

          if (currentAdIndex < adLinks.length) {
            setTimeout(showAdModal, 1000);
          } else {
            downloadLinks.forEach(link => {
              link.classList.add("enabled");
              link.style.background = "linear-gradient(45deg, #4CAF50, #81c784, #4CAF50)";
              link.style.backgroundSize = "200% 100%";
              link.style.opacity = "1";
            });
            
            gsap.killTweensOf(downloadLinks);
            if (disabledTween) disabledTween.kill();
            
            gsap.to(adModal, { 
              opacity: 0, 
              duration: 0.5, 
              onComplete: () => {
                adModal.classList.add("hidden");
                document.body.style.overflow = "auto";
                window.scrollTo(0, 0);
                animateEnabledButtons();
              }
            });
          }
        }
      }, 1000);
    }

    startButton.addEventListener("click", () => {
      gsap.to(startContainer, { opacity: 0, duration: 0.5, onComplete: () => startContainer.style.display = "none" });
      showAdModal();
    });

    window.addEventListener("scroll", function () {
      if (!adModal.classList.contains("hidden")) {
        window.scrollTo(0, 0);
      }
    });
  </script>
